How Phaser Plays a Scene Script (All Pre-Built)

// scene-script-player.ts â€” The director that assembles pre-built pieces

class SceneScriptPlayer {
  // Everything is pre-loaded in Phaser's preload()
  private actors: Map<string, Phaser.GameObjects.Sprite> = new Map();
  private props: Map<string, Phaser.GameObjects.Image> = new Map();

  // Position map â€” predefined screen coordinates
  private positions = {
    'left':      { x: 150, y: 400 },
    'center':    { x: 500, y: 400 },
    'right':     { x: 850, y: 400 },
    'top':       { x: 500, y: 150 },
    'bottom':    { x: 500, y: 550 },
    'off-left':  { x: -100, y: 400 },
    'off-right': { x: 1100, y: 400 },
    'off-top':   { x: 500, y: -100 },
  };

  // Move styles â€” predefined tween configs
  private moveStyles = {
    'linear':   { ease: 'Linear', duration: 600 },
    'arc':      { ease: 'Sine.easeInOut', duration: 800 },  // + y offset for arc
    'bounce':   { ease: 'Bounce.easeOut', duration: 700 },
    'float':    { ease: 'Sine.easeInOut', duration: 1200 },
    'shake':    { ease: 'Linear', duration: 300 },  // uses shake plugin
    'spin-in':  { ease: 'Back.easeOut', duration: 600 },
    'drop-in':  { ease: 'Bounce.easeOut', duration: 500 },
  };

  async executeAction(action: SceneAction): Promise<void> {
    switch (action.type) {
      case 'spawn': {
        // Look up pre-loaded texture. Place at named position.
        const pos = this.positions[action.position || 'center'];
        if (ACTOR_KEYS.includes(action.actor)) {
          const sprite = this.scene.add.sprite(pos.x, pos.y, action.actor);
          sprite.play(`${action.actor}-idle`); // Default to idle animation
          this.actors.set(action.actor, sprite);
        } else {
          // It's a prop (static image)
          const img = this.scene.add.image(pos.x, pos.y, action.actor);
          this.props.set(action.actor, img);
        }
        break;
      }

      case 'move': {
        const target = this.actors.get(action.actor) || this.props.get(action.actor);
        if (!target) return; // Graceful skip
        const dest = this.positions[action.to || 'center'];
        const style = this.moveStyles[action.style || 'linear'];
        return new Promise(resolve => {
          this.scene.tweens.add({
            targets: target,
            x: dest.x,
            y: dest.y,
            ...style,
            onComplete: () => resolve()
          });
        });
      }

      case 'animate': {
        const sprite = this.actors.get(action.actor);
        if (!sprite) return;
        const animKey = `${action.actor}-${action.anim}`;
        // Animation was registered in create() from the atlas
        return new Promise(resolve => {
          sprite.play(animKey);
          sprite.once('animationcomplete', () => resolve());
        });
      }

      case 'react': {
        // Fire Lottie overlay via EventBus â†’ React renders it
        sendToUI({
          type: 'PLAY_REACTION',
          effect: action.effect,
          position: action.position || 'center'
        });
        return this.wait(1200); // Let Lottie play
      }

      case 'remove': {
        const obj = this.actors.get(action.actor) || this.props.get(action.actor);
        if (obj) {
          this.scene.tweens.add({
            targets: obj, alpha: 0, duration: 300,
            onComplete: () => obj.destroy()
          });
        }
        return this.wait(300);
      }

      // ... emote, sfx, wait follow same pattern
    }
  }
}

Notice: Zero image generation. Zero dynamic rendering. Zero network calls. Every visual element was loaded in Phaser's preload(). The player just reads the script and places, moves, and animates pre-existing pieces. It's a puppet show, not a render pipeline.

function registerAnimations(scene: Phaser.Scene) {
  // Each actor atlas has tagged animations
  // These are created ONCE and available globally
  const actors = {
    'monster': ['idle', 'happy', 'sad', 'eat', 'dance', 'cry', 'sneeze-fire', 'confused', 'wave'],
    'dog': ['idle', 'bark', 'float', 'fetch', 'wag', 'helmet-on', 'spin', 'sleep'],
    'trex': ['idle', 'walk', 'stuck', 'blush', 'roar', 'sit-break', 'wave', 'pencil-try'],
    // ... etc
  };

  for (const [actor, anims] of Object.entries(actors)) {
    for (const anim of anims) {
      scene.anims.create({
        key: `${actor}-${anim}`,
        frames: scene.anims.generateFrameNames(actor, {
          prefix: `${anim}-`, start: 0, end: 5, zeroPad: 2
        }),
        frameRate: 8,
        repeat: anim === 'idle' ? -1 : 0  // idle loops, others play once
      });
    }
  }
}

async function generateMissingProp(description: string): Promise<string> {
  const response = await client.messages.create({
    model: 'claude-opus-4-6',
    max_tokens: 2000,
    messages: [{
      role: 'user',
      content: `Generate a simple SVG icon of: "${description}"
        Style: pixel-art inspired, flat colors, 64x64px viewport
        Return ONLY the SVG code, no explanation.`
    }]
  });

  const svg = response.content[0].text;
  // Convert SVG to image data URL
  const blob = new Blob([svg], { type: 'image/svg+xml' });
  const url = URL.createObjectURL(blob);

  // Add to Phaser's texture manager dynamically
